\section{Filling equation}

\begin{equation}
      u'= -\frac{P_su^2}{P_su^2+P_nv^2}/s
\end{equation}
\begin{equation}
      v'= -\frac{P_nv^2}{P_su^2+P_nv^2}/n
\end{equation}

we deduce

\begin{equation}
    \frac{u'}{u^2} - \frac{P_sn}{P_ns} \frac{v'}{v^2}= 0
\end{equation}
\begin{equation}
    p u'+n v' = -1
\end{equation}

with $ m = \frac{P_sn}{P_ns} $

it follows

\begin{equation}
    \frac{1}{u} - \frac{m}{v} = k
\end{equation}
\begin{equation}
    p u+n v = l-t
\end{equation}

Hence

\begin{equation}
    n(\frac{p}{n}t+k)u^2+((p+n)+(l-t)(\frac{p}{n}t+k))u-(l-t) = 0
\end{equation}

giving the solution

\begin{equation}
    v = \frac{(l-mn-p-t)-\sqrt{(l-mn-p-t)^2+k4n(t-l)}}{2kn}
\end{equation}
